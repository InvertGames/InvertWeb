Registering Instances
Getting Started > General uFrame Architecture > Dependency Injection > Registering Instances

Registering instances allow you to use an implementation that is already been manually initialized. 

Example Title
Copy Code 
using System;
public class SampleSceneManager : SceneManager
{
    public override void Setup()
    {
        base.Setup();
        // You can access by [Inject] attribute or container.Resolve<IMessenger>();
        Container.RegisterInstance<IMessenger>(new FacebookMessenger("asdf1234"));

        // This will register the instance under the name "Yahoo".
        // You can access by [Inject("Yahoo")] or container.Resolve<IMessenger>("Yahoo");
        Container.RegisterInstance<IMessenger>(new YahooMessenger("asdf1234"), "Yahoo");
        
    }
}
public class SampleController : Controller
{
    [Inject]
    public IMessenger Messenger { get; set; }
    [Inject("Yahoo")]
    public IMessenger YahooMessenger { get; set; }

    public void SendMessage(string message)
    {
        Messenger.SendMessage(DateTime.Now, message);
    }

    public void SendMessageToAll(string message)
    {
        // Resolve all will resolve all named instance excluding any direct instances like the Facebook messenger in this example.
        var messengers = GameManager.Container.ResolveAll<IMessenger>();
        foreach (var messenger in messengers)
             messenger.SendMessage(DateTime.Now, message);
    }
}
